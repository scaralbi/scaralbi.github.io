{{/* Syntax highlight a file of code * Arguments: * filename: A path relative to the calling page * language: A language supported by the Hugo syntax highlighter * options: (Optional) A list of options to pass to the Hugo syntax highlighter */}} {{- $filename := (path.Join (path.Dir .Page.File.Path) (.Get 0)) -}} {{- $language := .Get 1 -}} {{- $options := .Get 2 -}} {{- $fileContents := readFile $filename | safeHTML -}} {{- highlight $fileContents $language $options -}}
